function Vamp.Rituals.AwaitReturn()
local waitStart=tick()
while getgenv().VampActive do
if tick()-waitStart > Vamp.Grimoire.Law.ReturnTimeout then 
Vamp.Rituals.Dissipate(true); break 
end
local char=LP.Character;local hrp=char and char:FindFirstChild("HumanoidRootPart");local hum=char and char:FindFirstChild("Humanoid")
if hrp and hum then 
local dist = (hrp.Position - Vamp.Grimoire.Law.LobbyCenter).Magnitude
if dist <= Vamp.Grimoire.Law.LobbyReturnProximity and hum.FloorMaterial~=Enum.Material.Air then 
Vamp.Grimoire.Law.ScanningPos = hrp.CFrame
task.wait(2)
break 
end 
end
task.wait()
end
end

local hrp = LP.Character.HumanoidRootPart
local dist = (hrp.Position - Vamp.Grimoire.Law.ScanningPos.Position).Magnitude

hrp.CFrame = Vamp.Grimoire.Law.ScanningPos
Vamp.Grimoire.Trance.IsAtScanningPos = true
end

Vamp.Grimoire.Trance.IsAtScanningPos = true
end
end
end

local function GetSpotScore(spot)
local allowedIndices = Vamp.Grimoire.Desires.ROLE == "WINNER" and {1,3} or {2,4}
local isAllowed = false
for _, validIdx in ipairs(allowedIndices) do if spot.Index == validIdx then isAllowed = true break end end
if not isAllowed then return -100 end
local isIntruded = (spot.Occupied.Top > 0 or spot.Partner.Occupied.Top > 0)
if isIntruded then
if spot == Trance.CurrentAltar then
if Trance.IntrusionStartTime == 0 then Trance.IntrusionStartTime = tick() end
if tick() - Trance.IntrusionStartTime > (Law.IntrusionBuffer) then
return -100 
end
else
return -100 
end
else
if spot == Trance.CurrentAltar then Trance.IntrusionStartTime = 0 end
end
local amIPresent = false
for _, name in ipairs(spot.Players.Bot) do
if name == LP.Name then amIPresent = true break end
end
local myCount = spot.Occupied.Bot - (amIPresent and 1 or 0)
if myCount < 0 then myCount = 0 end 
local partnerCount = spot.Partner.Occupied.Bot
if Vamp.Grimoire.Desires.MODE == "2V2" then
if Vamp.Grimoire.Desires.ROLE == "WINNER" then
if Vamp.Grimoire.Desires.TIER == "PRIORITY" then
if myCount > 0 then return -100 end 
if partnerCount >= 2 then return 100 
elseif partnerCount == 1 then return 90 
elseif partnerCount == 0 then return 80 
end
else
if myCount ~= 1 then return -100 end 
if partnerCount >= 2 then return 100 
elseif partnerCount == 1 then return 90 
elseif partnerCount == 0 then return 80 
end
end
else
if myCount >= 2 then return -100 end
if myCount == 1 and partnerCount >= 2 then return 100 
elseif myCount == 0 and partnerCount >= 2 then return 90 
elseif myCount == 1 and partnerCount == 1 then return 80 
elseif myCount == 0 and partnerCount == 1 then return 70 
elseif myCount == 0 and partnerCount == 0 then return 60 
else return 50 end
end
else
if Vamp.Grimoire.Desires.ROLE == "WINNER" then
if myCount > 0 then return -100 end
if partnerCount >= 1 then return 100 else return 90 end
else
if myCount > 0 then return -100 end
if partnerCount >= 1 then return 100 else return 90 end
end
end
return -100
end
local bestSpot = nil
local bestScore = -100
local currentScore = -100
if Trance.CurrentAltar then
currentScore = GetSpotScore(Trance.CurrentAltar)
elseif Trance.PhysicalAltar then
currentScore = GetSpotScore(Trance.PhysicalAltar)
end
for _, spot in ipairs(Altars) do
local score = GetSpotScore(spot)
if score > bestScore then
bestScore = score
bestSpot = spot
end
end
if bestScore == -100 then
if Trance.CurrentAltar or Trance.PhysicalAltar then
Vamp.Rituals.Dissipate(true)
Trance.CurrentAltar = nil
Trance.IsAtScanningPos = true
Vamp.Visage.Refresh("SCANNING")
else
if not Trance.IsAtScanningPos then
Vamp.Rituals.Dissipate(true)
Trance.IsAtScanningPos = true
Vamp.Visage.Refresh("SCANNING")
end
end
else
if currentScore == -100 or bestScore > currentScore then
if bestSpot == Trance.PhysicalAltar then
Trance.CurrentAltar = bestSpot
elseif bestSpot ~= Trance.CurrentAltar then
Vamp.Rituals.ShadowStep(bestSpot)
end
end
local function CheckLoading(spot)
if not spot then return false end
local count = spot.Occupied.Bot 
local pCount = spot.Partner.Occupied.Bot
if Vamp.Grimoire.Desires.MODE == "2V2" then
return (count >= 2 and pCount >= 2)
else
return (count >= 1 and pCount >= 1)
end
end
if bestSpot == Trance.CurrentAltar and CheckLoading(bestSpot) then

Vamp.Visage.Refresh("LOADING")

if Vamp.Grimoire.Desires.TIER == "PRIORITY" then

end
end
else
if bestSpot.Partner.Players.Bot[1] then
Vamp.Grimoire.Trance.TargetNames = {bestSpot.Partner.Players.Bot[1]}
end
end
else
Trance.LoadingStartTime = 0
Vamp.Visage.Refresh("IDLE")
end
end
end
